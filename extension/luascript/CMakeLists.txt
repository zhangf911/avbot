		luabind/src/class_rep.cpp

		luabind/src/object_rep.cpp
		luabind/src/scope.cpp
		luabind/src/function.cpp

		)

    $<TARGET_PROPERTY:xmpp,INTERFACE_INCLUDE_DIRECTORIES>)
